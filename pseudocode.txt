	PROCEDURE displayTodoList()
		STORE "i" with number 0
		WHILE length of "todo_list" > "i"
			DISPLAY "todo_list"["i"]
			ADD "i" by 1
	END PROCEDURE 
	
	FUNCTION delete(data)
		STORE "i" with number 0
		WHILE length of "todo_list" > "i"
			IF "todo_list"["i"] equal to "data"
				DELETE "todo_list"["i"] from "todo_list"
				RETURN "todo_list"
			ADD "i" by 1
		RETURN DISPLAY "Data Not Found"
	END FUNCTION
